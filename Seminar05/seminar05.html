#g: the selected genes
prepareData  <- function(g){
  #prDes row + gExp of the gene from prDat + gene from the input
  pDat <- data.frame()
  for (i in 1:length(g)){
    pDat1 <- data.frame(prDes, gExp = as.vector(as.matrix(prDat[g[i], ])), gene = g[i])
    pDat <- rbind(pDat, pDat1)
  } 
  pDat
}
